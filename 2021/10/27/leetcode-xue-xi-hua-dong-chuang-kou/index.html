什么是滑动窗口

滑动窗口也就是之前说过的双指针技巧，通过维护一个窗口，不断滑动，然后更新结果，其主要的应用场景就是在字符串的匹配上了，比如最小覆盖子串、最长无重复子串等。

滑动窗口的算法上确实维护着一个窗口，可实际理解起来还是有点麻烦，不过按照步骤划分的话也就如下的几步：

  • 首先我们依旧通过声明双指针 left = right = 0。
  • right ++ ，不断遍历源字符串，直到遍历过的字符串（窗口）中已经找到覆盖需要匹配的字符串要求(或者满足某种条件)
  • 然后 left++，缩小【left,right）的窗口，直到窗口中的字符串不符合匹配的字符串要求
  • 重复 2 和 3，直到遍历完整个字符串

最短覆盖子串

输入：s = "ADOBECODEBANC", t = "ABC"
输出："BANC"
// 滑动窗口（最小覆盖子串）  “AD0BEC0DEBANC”  “ABC”
public static string MinWindow(string srcStr, string patter)
{
    // 存储 需要识别的子串（A:1,B:1,C:1）
    Dictionary<char, int> need = new Dictionary<char, int>();
    // 滑动窗口中的字符串(A:2,B:1,C:2)
    Dictionary<char, int> window = new Dictionary<char, int>();
    // 拿到需要识别的子串数据
    foreach (char c in patter)
    {
        if (need.ContainsKey(c))
        {
            need[c]++;
        }
        else
        {
            need.Add(c, 1);
        }
    }

    int left = 0;
    int right = 0;
    int valid = 0;
    int start = 0;
    int len = int.MaxValue;
    while(right < srcStr.Length)
    {
        char c = srcStr[right];
        // 右移窗口
        right++;
        // 获取窗口中的符合 识别子串的字符进行缓存
        if(need.ContainsKey(c))
        {
            if (window.ContainsKey(c))
            {
                window[c]++;
            }
            else
            {
                window.Add(c, 1);
            }
            // 符合一个就计数1个
            if (window[c] == need[c])
            {
                valid++;
            }
        }

        // 这个时候已经找到了覆盖子串的字符串窗口进行左移缩小窗口
        while(valid == need.Count)//存在重复的可能
        {
            if(right - left < len)
            {
                start = left;
                len = right - left;
            }
            char d = srcStr[left];
            left++;
            if (need.ContainsKey(d))
            {
                if (window[d] == need[d])
                    valid--;
                window[d]--;
            }
        }
    }

    return len == int.MaxValue ? "" : srcStr.Substring(start,len);
}

字符串排列顺序

输入：s1 = "ab" s2 = "eidbaooo"
输出：true
解释：s2 包含 s1 的排列之一 ("ba").
public static bool checkInclusion(string t, string s)
{
    // 存储 需要识别的子串（A:1,B:1,C:1）
    Dictionary<char, int> need = new Dictionary<char, int>();
    // 滑动窗口中的字符串(A:2,B:1,C:2)
    Dictionary<char, int> window = new Dictionary<char, int>();
    // 拿到需要识别的子串数据
    foreach (char c in t)
    {
        if (need.ContainsKey(c))
        {
            need[c]++;
        }
        else
        {
            need.Add(c, 1);
        }
    }

    int left = 0;
    int right = 0;
    int valid = 0;
    bool isExist = false;
    while (right < s.Length)
    {
        char c = s[right];
        // 右移窗口
        right++;
        // 获取窗口中的符合 识别子串的字符进行缓存
        if (need.ContainsKey(c))
        {
            if (window.ContainsKey(c))
            {
                window[c]++;
            }
            else
            {
                window.Add(c, 1);
            }
            // 符合一个就计数1个
            if (window[c] == need[c])
            {
                valid++;
            }
        }

        // 这个时候已经找到了覆盖子串的字符串窗口进行左移缩小窗口
        while ((right -left)>=t.Length)
        {
            if (valid == need.Count)
            {
                isExist = true;
                break;
            }

            char d = s[left];
            left++;
            if (need.ContainsKey(d))
            {
                if (window[d] == need[d])
                    valid--;
                window[d]--;
            }
        }
    }

    return isExist;
}

最长无重复子串

输入: s = "abcabcbb"
输出: 3 
解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。
public static int LengthOfLongestSubstring(string s)
{
    Dictionary<char, int> window = new Dictionary<char, int>();

    int left = 0, right = 0;
    int res = 0; // 记录结果
    while (right < s.Length)
    {
        char c = s[right];
        right++;
        // 进行窗口内数据的一系列更新
        if (window.ContainsKey(c))
        {
            window[c]++;
        }
        else
        {
            window.Add(c, 1);
        }
        // 判断左侧窗口是否要收缩
        while (window.ContainsKey(c) && window[c] > 1)
        {
            char d = s[left];
            left++;
            // 进行窗口内数据的一系列更新
            window[d]--;
        }
        // 在这里更新答案
        if ((right - left) > res)
        {
            res = right - left;
        }
    }
    return res;
}
